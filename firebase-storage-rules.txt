// Règles Firebase Storage à configurer dans la console Firebase
// Allez sur https://console.firebase.google.com
// Sélectionnez votre projet misterpips-b71fb
// Allez dans Storage > Rules
// Remplacez les règles par ceci :

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}

// IMPORTANT: Ces règles permettent l'accès complet.
// Pour la production, utilisez des règles plus restrictives :

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /posts/{imageId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /gallery/{imageId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /avatars/{imageId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
